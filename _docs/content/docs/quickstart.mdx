# Quick Start

This guide will walk you through building a complete REST API with Surf in just a few minutes.

## Create a New Project

```bash
mkdir myapi && cd myapi
go mod init myapi
go get github.com/getangry/surf
```

## Build a REST API

Create `main.go` with a complete CRUD API:

```go title="main.go"
package main

import (
    "encoding/json"
    "net/http"
    "sync"

    "github.com/getangry/surf"
)

// User represents a user in our system
type User struct {
    ID    string `json:"id"`
    Name  string `json:"name"`
    Email string `json:"email"`
}

// In-memory store (use a real database in production)
var (
    users   = make(map[string]User)
    usersMu sync.RWMutex
)

func main() {
    app := surf.NewApp()

    // Add middleware
    app.Use(surf.RecoveryWithDefaults())
    app.Use(surf.CORSWithDefaults())
    app.Use(surf.RateLimitWithDefaults())

    // Health check
    app.Get("/health", healthHandler)

    // User routes
    api := app.Group("/api/v1")
    api.Get("/users", listUsers)
    api.Get("/users/:id", getUser)
    api.Post("/users", createUser)
    api.Put("/users/:id", updateUser)
    api.Delete("/users/:id", deleteUser)

    // Start server
    app.Serve()
}

func healthHandler(w http.ResponseWriter, r *http.Request) error {
    return json.NewEncoder(w).Encode(map[string]string{
        "status": "healthy",
    })
}

func listUsers(w http.ResponseWriter, r *http.Request) error {
    // Query parameters for pagination
    page := surf.QueryInt(r, "page", 1)
    limit := surf.QueryInt(r, "limit", 10)

    usersMu.RLock()
    defer usersMu.RUnlock()

    result := make([]User, 0, len(users))
    for _, u := range users {
        result = append(result, u)
    }

    w.Header().Set("Content-Type", "application/json")
    return json.NewEncoder(w).Encode(map[string]interface{}{
        "users": result,
        "page":  page,
        "limit": limit,
    })
}

func getUser(w http.ResponseWriter, r *http.Request) error {
    id := surf.Param(r, "id")

    usersMu.RLock()
    user, exists := users[id]
    usersMu.RUnlock()

    if !exists {
        http.Error(w, "User not found", http.StatusNotFound)
        return nil
    }

    w.Header().Set("Content-Type", "application/json")
    return json.NewEncoder(w).Encode(user)
}

func createUser(w http.ResponseWriter, r *http.Request) error {
    var user User
    if err := json.NewDecoder(r.Body).Decode(&user); err != nil {
        http.Error(w, "Invalid JSON", http.StatusBadRequest)
        return nil
    }

    usersMu.Lock()
    users[user.ID] = user
    usersMu.Unlock()

    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(http.StatusCreated)
    return json.NewEncoder(w).Encode(user)
}

func updateUser(w http.ResponseWriter, r *http.Request) error {
    id := surf.Param(r, "id")

    var user User
    if err := json.NewDecoder(r.Body).Decode(&user); err != nil {
        http.Error(w, "Invalid JSON", http.StatusBadRequest)
        return nil
    }

    usersMu.Lock()
    if _, exists := users[id]; !exists {
        usersMu.Unlock()
        http.Error(w, "User not found", http.StatusNotFound)
        return nil
    }
    user.ID = id
    users[id] = user
    usersMu.Unlock()

    w.Header().Set("Content-Type", "application/json")
    return json.NewEncoder(w).Encode(user)
}

func deleteUser(w http.ResponseWriter, r *http.Request) error {
    id := surf.Param(r, "id")

    usersMu.Lock()
    delete(users, id)
    usersMu.Unlock()

    w.WriteHeader(http.StatusNoContent)
    return nil
}
```

## Test Your API

Run the server:

```bash
go run main.go
```

Test with curl:

```bash
# Create a user
curl -X POST http://localhost:8080/api/v1/users \
  -H "Content-Type: application/json" \
  -d '{"id":"1","name":"John Doe","email":"john@example.com"}'

# List users
curl http://localhost:8080/api/v1/users

# Get a specific user
curl http://localhost:8080/api/v1/users/1

# Update a user
curl -X PUT http://localhost:8080/api/v1/users/1 \
  -H "Content-Type: application/json" \
  -d '{"name":"Jane Doe","email":"jane@example.com"}'

# Delete a user
curl -X DELETE http://localhost:8080/api/v1/users/1
```

## What's Included

This example demonstrates:

- **Routing**: Path parameters (`:id`) and route groups (`/api/v1`)
- **Middleware**: Recovery, CORS, and rate limiting
- **Query Helpers**: Type-safe query parameter parsing
- **JSON Handling**: Request/response encoding
- **Thread Safety**: Proper mutex usage for concurrent access
